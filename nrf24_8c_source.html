37 /* configure the module */
38 void nrf24_config(const uint8_t channel, const uint8_t pay_length)
39 {
40  /* Use static payload length ... */
41  Nrf24.payload_len = pay_length;
42 
43  // Set RF channel
44  nrf24_configRegister(RF_CH,channel);
45 
46  // Set length of incoming payload
47  nrf24_configRegister(RX_PW_P0, 0x00); // Auto-ACK pipe ...
48  nrf24_configRegister(RX_PW_P1, Nrf24.payload_len); // Data payload pipe
49  nrf24_configRegister(RX_PW_P2, 0x00); // Pipe not used
50  nrf24_configRegister(RX_PW_P3, 0x00); // Pipe not used
51  nrf24_configRegister(RX_PW_P4, 0x00); // Pipe not used
52  nrf24_configRegister(RX_PW_P5, 0x00); // Pipe not used
53 
54  // 1 Mbps, TX gain: 0dbm
55  nrf24_configRegister(RF_SETUP, (0<<RF_DR)|((0x03)<<RF_PWR));
56 
57  // CRC enable, 1 byte CRC length
58  nrf24_configRegister(CONFIG,nrf24_CONFIG);
59 
60  // Auto Acknowledgment
61  nrf24_configRegister(EN_AA,_BV(ENAA_P0)|_BV(ENAA_P1)|(0<<ENAA_P2)|(0<<ENAA_P3)|(0<<ENAA_P4)|(0<<ENAA_P5));
62 
63  // Enable RX addresses
64  nrf24_configRegister(EN_RXADDR,_BV(ERX_P0)|_BV(ERX_P1)|(0<<ERX_P2)|(0<<ERX_P3)|(0<<ERX_P4)|(0<<ERX_P5));
65 
66  // Auto retransmit delay: 1000 us and Up to 15 retransmit trials
67  nrf24_configRegister(SETUP_RETR,(0x04<<ARD)|(0x0F<<ARC));
68 
69  // Dynamic length configurations: No dynamic length
70  nrf24_configRegister(DYNPD,(0<<DPL_P0)|(0<<DPL_P1)|(0<<DPL_P2)|(0<<DPL_P3)|(0<<DPL_P4)|(0<<DPL_P5));
71 
72  // Start listening
73  nrf24_powerUpRx();
74 }

98 /* Checks if data is available for reading */
99 /* Returns 1 if data is ready ... */
100 bool nrf24_dataReady()
101 {
102  // See note in getData() function - just checking RX_DR isn't good enough
103  const uint8_t status = nrf24_getStatus();
104 
105  // We can short circuit on RX_DR, but if it's not set, we still need
106  // to check the FIFO for any pending packets
107  if ( status & _BV(RX_DR) )
108  {
109  return true;
110  }
111 
112  return !nrf24_rxFifoEmpty();;
113 }

136 /* Reads payload bytes into data array */
137 void nrf24_getData(uint8_t* const data)
138 {
139  /* Pull down chip select */
140  nrf24_csn_digitalWrite(LOW);
141 
142  /* Send cmd to read rx payload */
143  Spi.transfer(R_RX_PAYLOAD);
144 
145  /* Read payload */

147 
148  /* Pull up chip select */

151  /* Reset status register */
152  nrf24_configRegister(STATUS,_BV(RX_DR));

164 // Sends a data package to the default address. Be sure to send the correct
165 // amount of bytes as configured as payload on the receiver.
166 void nrf24_send(const uint8_t* const value)
167 {
168  /* Go to Standby-I first */
169  nrf24_ce_digitalWrite(LOW);
170 
171  /* Set to transmitter mode , Power up if needed */
172  nrf24_powerUpTx();
173 
174  /* Do we really need to flush TX fifo each time ? */
175  #if 1
176  /* Pull down chip select */
177  nrf24_csn_digitalWrite(LOW);
178 
179  /* Write cmd to flush transmit FIFO */
180  Spi.transfer(FLUSH_TX);
181 
182  /* Pull up chip select */
183  nrf24_csn_digitalWrite(HIGH);
184  #endif
185 
186  /* Pull down chip select */
187  nrf24_csn_digitalWrite(LOW);
188 
189  /* Write cmd to write payload */
190  Spi.transfer(W_TX_PAYLOAD);
191 
192  /* Write payload */
193  Spi.multiTransfer(value,NULL,Nrf24.payload_len);
194 
195  /* Pull up chip select */
196  nrf24_csn_digitalWrite(HIGH);
197 
198  /* Start the transmission */
199  nrf24_ce_digitalWrite(HIGH);
200 }

202 bool nrf24_isSending()
203 {
204  uint8_t status;
205 
206  /* read the current status */
207  status = nrf24_getStatus();
208 
209  /* if sending successful (TX_DS) or max retries exceded (MAX_RT). */
210  if((status & ((1 << TX_DS) | (1 << MAX_RT))))
211  {
212  return false; /* false */
213  }
214 
215  return true; /* true */
216 
217 }

228 uint8_t nrf24_lastMessageStatus()
229 {
230  uint8_t rv;
231 
232  rv = nrf24_getStatus();
233 
234  /* Transmission went OK */
235  if((rv & ((1 << TX_DS))))
236  {
237  return NRF24_TRANSMISSON_OK;
238  }
239  /* Maximum retransmission count is reached */
240  /* Last message probably went missing ... */
241  else if((rv & ((1 << MAX_RT))))
242  {
243  return NRF24_MESSAGE_LOST;
244  }
245  /* Probably still sending ... */
246  else
247  {
248  return 0xFF;
249  }
250 }
