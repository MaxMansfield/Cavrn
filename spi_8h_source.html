32 /* SPI_MODEn
33  * The bit mask of the CPOL and CPHA bits which can be directly used
34  * to set SPCR
35  */
36 #define SPI_MODE0 0xf3 // CPOL = 0; CPHA = 0; (use AND not OR)
37 #define SPI_MODE1 0x04 // CPOL = 0; CPHA = 1;
38 #define SPI_MODE2 0x08 // CPOL = 1; CPHA = 0;
39 #define SPI_MODE3 0x0C // CPOL = 1; CPHA = 1;

41 /* SCK rate control.
42  * This also directly effects the Oscillator Clock frequency, so it is refered to as F_OSC.
43  * Each clock value is the bitmask of what needs to be put into the SPCR. SPI2X F_OSC speeds
44  * are defined as necessary below.
45  */
46 #define F_OSC4 0x00
47 #define F_OSC16 (_BV(SPR0))
48 #define F_OSC64 (_BV(SPR1))
49 #define F_OSC128 (_BV(SPR1) | _BV(SPR0))
50 
51 
52 /* SPI2X
53  * Check for the SPI2X register. If available then define the speeds (F_OSC64 is only defined once)
54  * If it's not available then define the functions as nothing so that the finished program has just
55  * a semicolon which is the nop instruction in C.
56  */
57 #ifdef SPI2X
58 #define F_OSC2 0x00 // Will not change the bits, bit will already be clear
59 #define F_OSC8 (_BV(SPR0)) // Set only one bit
60 #define F_OSC32 (_BV(SPR1)) // Set only one bit
61 #define F_OSC64x (_BV(SPR1) | _BV(SPR0)) // Set both bits
