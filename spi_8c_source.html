27 void spi_init(const bool isMaster,const byte mode,const uint8_t clk)
28 {
29 
30  SPI_DDR |= (_BV(SPI_SCK) | _BV(SPI_SS));
31  SPI_PORT |= (_BV(SPI_SCK) | _BV(SPI_SS));
32  SPCR |= _BV(SPE);
33  if(isMaster){
34 
35  // MOSI=out MISO=in SCK=out SS=out
36  SPI_DDR |= _BV(SPI_MOSI) | _BV(SPI_SCK) | _BV(SPI_SCK);
37  SPI_PORT |= _BV(SPI_MOSI) | _BV(SPI_SCK) | _BV(SPI_SCK);
38 
39  SPI_DDR &= ~_BV(SPI_MISO);
40  SPI_PORT &= ~_BV(SPI_MISO);
41 
42  ENABLE_MASTER();
43 
44  /* Only show if/else when SPI2X is present */
45 
46 #if defined( __AVR_ATmega328P__) || defined(SPI2X)
47  if(Spi.useSpi2x){
48  switch(clk){
49  case (F_OSC2):
50  case (F_OSC8):
51  case (F_OSC32):
52  case (F_OSC64x):
53  SET_SPI2X();
54  CLR_SCK_REG();
55  SPCR |= clk;
56  break;
57  default:
58  SET_SPI2X(); // Use SPI2X
59  CLR_SCK_REG(); // Clear both bits to 0
60  SPCR |= F_OSC8; // Set SPR0 to 1
61  break;
62  }
63  }
64  else
65 #endif
66  {
67  switch(clk){
68  case (F_OSC4):
69  case (F_OSC16):
70  case (F_OSC64):
71  case (F_OSC128):
72  CLR_SPI2X();
73  CLR_SCK_REG();
74  SPCR |= clk;
75  break;
76  default:
77  CLR_SPI2X(); // Don't use spi2x
78  CLR_SCK_REG();
79  SPCR |= F_OSC16;
80  break;
81  }
82  }
83  } else {
84  //Enter Slave mode
85  // MOSI=in MISO=out SCK=in SS=in
86  SPI_DDR |= (_BV(SPI_MISO));
87  SPI_PORT |= (_BV(SPI_MISO));
88 
89  SPI_DDR &= ~(_BV(SPI_MOSI) | _BV(SPI_SCK) | _BV(SPI_SCK) );
90  SPI_PORT &= ~(_BV(SPI_MOSI) | _BV(SPI_SCK) | _BV(SPI_SCK) );
91  ENABLE_SLAVE();
92  }
93 
94  switch(mode){
95  case (SPI_MODE0):
96  INIT_SPI_MODE0();
97  break;
98  case (SPI_MODE1):
99  INIT_SPI_MODE1();
100  break;
101  case(SPI_MODE2):
102  INIT_SPI_MODE2();
103  break;
104  case(SPI_MODE3):
105  INIT_SPI_MODE3();
106  break;
107  default:
108  INIT_SPI_MODE0();
109  break;
110  }
111  return;
112 }

149 {
150  //Disable interrupts during transfer

152 
153  if(dataout){

155  byte sreg = SREG;
156  bool intr;

158  Spi.setInterrupts(OFF);
159 
160  for(i=0;i<len;i++)
161  {
162  if(datain)
163  datain[i] = spi_transfer(dataout[i]);
164  else
165  spi_transfer(dataout[i]);
166  }
167 
168  SREG = sreg;
169 
170  //Enable interrupts if they were disabled
171  if(intr)
172  Spi.setInterrupts(ON);
173  }
174 
175 
176  return;
